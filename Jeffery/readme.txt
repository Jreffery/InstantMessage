{
	"type":8000,         //请求码
	"port":8010          //端口号，暂定
}

//响应
{
	"code":200,          //200成功，503错误
	"errMsg":null        //返回信息，如果有
}

{
	"type":8001,
	"appid":'1111111111111111'       //16位
}

//响应
{
	"code":200,         //200成功，503无服务
	"nodeMsg":'127.0.0.1-8010'     //节点信息
}
